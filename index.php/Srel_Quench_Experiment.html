import subprocess
import os

import threading

def Write_Script(Tinitial,Tfinal):
	global text, runjob
	text1=text
	text1=text1.replace('XTinitial',Tinitial)
	text1=text1.replace('XTfinal',Tfinal)
	file("script.txt","w").write(text1)
	file("script.txt").close()
	file('submitjob.sh','w').write(runjob)
	file('submitjob.sh').close()

def CalcT2(Tf):
	global j, k
	Ti=float(k)
	Srel=float(j)
	a=1444.47775
	b=0.609113464
	y=a/Ti**(1-b)*(b/(1-b)+Ti/Tf-1/(1-b)*(Ti/Tf)**(1-b))-Srel
	return y

def CalcSrel(Ti,Tf):
	a=1444.47775
	b=0.609113464
	y=a/Ti**(1-b)*(b/(1-b)+Ti/Tf-1/(1-b)*(Ti/Tf)**(1-b))
	return y

def runk():
	global Paths, Pathslock, rootdir, chdirlock, chdirlockhistory
	runkdata=threading.local()
	while len(Paths) != 0:
		Pathslock.acquire()
		runkdata.path=Paths[0]
		Paths.pop(0)
		Pathslock.release()
		runkdata.path=runkdata.path.split('/s')
		chdirlock.acquire()
		chdirlockhistory.append('lock opened %(0)s by %(1)s for job submission'%{'0':time.asctime(),'1':runkdata.path[0]})
		os.chdir(rootdir)
		os.chdir(runkdata.path[0])
		os.popen("ssh zin.local 'cd %(0)s;cd %(1)s;qsub submitjob.sh >jobnum.txt'"%{'0':rootdir,'1':runkdata.path[0]})
		time.sleep(5)
		runkdata.num=file('jobnum.txt','r').read().split(' ')[2]
		ev1 = threading.Event()
		t1 = threading.Thread(target=repeat, args=(ev1, 30.0, querryqstat,runkdata.num,runkdata.path))
		print("starting thread %s" %runkdata.path[0])
		t1.start()
		os.chdir(rootdir)
		chdirlockhistory.append('lock closed %(0)s by %(1)s'%{'0':time.asctime(),'1':runkdata.path[0]})
		chdirlock.release()
		t1.join()
	
	print('Job Queue has beed depleted %s killed'%threading.current_thread())

def repeat(event, every, action, num,path): 
	mydata=threading.local()
	while True: 
		event.wait(every) 
		if event.isSet(): 
			break 
		action(event,num,path) 

def setqstat(event,num,path):
	global numlist, numlistlock,chdirlock
	numlistlock.acquire()
	numlist=str(os.popen('qstat').readlines())
	numlistlock.release()

def querryqstat(event,num,path):
	global numlist, numlistlock
	mydata=threading.local()
	numlistlock.acquire()
	mydata.numexists=numlist.find(num)
	numlistlock.release()
	if mydata.numexists == -1:
		print('job %s complete...analyzing' %num)		
		analyze(event,num,path)		
	

def analyze(event,num,path):
	global Pathslock, Paths,rootdir,chdirlock,chdirlockhistory
	chdirlock.acquire()
	chdirlockhistory.append('lock opened %(0)s by %(1)s for analysis'%{'0':time.asctime(),'1':path[0]})
	os.chdir(rootdir)
	os.chdir(path[0])
	Tinitial=float(os.getcwd().split('/')[-2])
	Tfinal=float(os.getcwd().split('/')[-1].split('_')[-1])
	os.popen("sed '1,%(0)sd;%(1)s,%(2)sd' log.txt > log1.txt; awk  '{print $6}' log1.txt > log2.txt"%{'0':str(numeq+79),'1':str(numeq+80+numprod),'2':str(numeq+104+numprod)})
	time.sleep(3)
	Text=file('log2.txt','r').read()
	Text=Text.rstrip()
	Text=Text.split('\n')
	Text=[float (o) for o in Text]
	for k in Text:
		file('average.txt','a').write('%.7f\n'%k)
	
	avg=file('average.txt','r').read()
	avg=avg.rstrip()
	avg=avg.split('\n\n')
	length=len(avg[0].split('\n'))
	AVG=np.zeros(length,float)
	for k in avg:
		temp=[float(l) for l in k.split('\n')]
		AVG=AVG+np.array(temp,float)
	
	size=len(avg)
	AVG=AVG/size
	file('average.txt','a').write('\n')
	v=[ -4.87425549e+03,   1.44447775e+03,   6.09113464e-01]
	ft = lambda v, T: v[0]+v[1]*T**v[2] 
	Text1=[np.add(n,-ft(v,Tfinal)) for n in AVG]
	Text2=[np.divide(m,(ft(v,Tinitial)-ft(v,Tfinal))) for m in Text1]
	fp = lambda z, x: z[0]*np.exp(-(z[1]*x+z[2]*x**0.5+z[3]*x**0.3))
	e = lambda z, x, y: (fp(z,x)-y)
	y = Text2
	x = np.linspace(1,len(y),len(y))
	z0 = [1, 1, 1,1]
	z, success = scipy.optimize.leastsq(e, z0, args=(x,y), maxfev=10000)
	stderr=np.sum((e(z,x,y))**2.0)/len(y)
	if stderr >= 0.0001:
		Pathslock.acquire()
		print('resubmitting %s'%path)
		Paths.insert(0,'./%(0)s/%(1)s/submitjob.sh'%{'0':os.getcwd().split('/')[-2],'1':os.getcwd().split('/')[-1]})
		Pathslock.release()
		os.popen('rm SrelExp*')
		event.set()
	
	if stderr <=0.0001:
		print('%.4f'%stderr)
		Text=file("average.txt","r").read()
		Text=Text.rstrip()
		Text=Text.split("\n\n")
		Length=len(Text[0].split("\n"))
		Scott=np.zeros(Length,float)
		for Block in Text:
        		Avi=[float (k) for k in Block.split("\n")]
        		Scott = Scott + np.array(Avi,float)
		
		Size=len(Text)
		Scott = Scott / Size
		file("logavg.xls","w").write("Data\n")
		for k in Scott:
		        file("logavg.xls","a").write("%.4f\n"%k)
		
		event.set()
	
	os.chdir(rootdir)
	chdirlockhistory.append('lock closed %(0)s by %(1)s'%{'0':time.asctime(),'1':path[0]})
	chdirlock.release()
